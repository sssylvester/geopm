152  template <class type>
153  void CircularBuffer<type>::set_capacity(const unsigned int size)
154  {
155  if (size < m_count) {
156  int size_diff = m_count - size;
157  std::vector<type> temp;
158  //Copy newest data into temporary vector
159  for (unsigned int i = m_head + size_diff; i < ((m_head + m_count) % m_max_size); i = ((i + 1) % m_max_size)) {
160  temp.push_back(m_buffer[i]);
161  }
162  //now re-size and swap out with tmp vector data
163  m_buffer.resize(size);
164  m_buffer.swap(temp);
165  m_count = size;
166  }
167  else {
168  m_buffer.resize(size);
169  }
170  m_head = 0;
171  m_max_size = size;
172  }

174  template <class type>
175  void CircularBuffer<type>::insert(const type value)
176  {
177  if (m_max_size < 1) {
178  throw Exception("CircularBuffer::insert(): Cannot insert into a buffer of 0 size", GEOPM_ERROR_RUNTIME, __FILE__, __LINE__);
179  }
180  if (m_count < m_max_size) {
181  m_buffer.push_back(value);
182  m_count++;
183  }
184  else {
185  m_buffer[m_head] = value;
186  m_head = ((m_head + 1) % m_max_size);
187  }
188  }
